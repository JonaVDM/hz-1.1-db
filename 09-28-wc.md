## 6.13
```sql
SELECT NR, OMSCHRIJVING, ARTIKELGROEP FROM (
	SELECT a.NR, a.OMSCHRIJVING, a2.OMSCHRIJVING AS ARTIKELGROEP FROM ARTIKEL a
	JOIN ARTIKELGROEP a2 ON a.ARTIKELGROEP = a2.CODE 

	UNION 

	SELECT NR, OMSCHRIJVING, 'los artikel' AS ARTIKELGROEP FROM ARTIKEL a 	
	WHERE ARTIKELGROEP IS NULL 
)
ORDER BY ARTIKELGROEP, NR
```
